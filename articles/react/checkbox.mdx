## Overview

The `Checkbox` component provides an accessible toggle input built on WAI-ARIA checkbox pattern. It supports render props for full visual customization, indeterminate state, and works seamlessly with form libraries.

## Usage

<CodeSample
  filename="Basic.tsx"
  language="tsx"
  code={`import { Checkbox } from '@sprocketui/react';

function App() {
  const [checked, setChecked] = useState(false);

  return (
    <Checkbox
      isSelected={checked}
      onChange={setChecked}
    >
      Accept terms
    </Checkbox>
  );
}`}
/>

<CodeSample
  filename="Indeterminate.tsx"
  language="tsx"
  code={`import { Checkbox } from '@sprocketui/react';

function SelectAll({ items, selected, onToggleAll }) {
  const allSelected = selected.length === items.length;
  const someSelected = selected.length > 0 && !allSelected;

  return (
    <Checkbox
      isSelected={allSelected}
      isIndeterminate={someSelected}
      onChange={onToggleAll}
    >
      Select all
    </Checkbox>
  );
}`}
/>
